#include <iostream>
using namespace std;

int main() {
    int n, m;
    cin >> n;
    m = n % 100;

    if (m>=3 && m<=5)
        cout << "spring";
    else if (m>=6 && m<=8)
        cout << "summer";
    else if (m>=9 && m<=11)
        cout << "autumn";
    else 
        cout << "winter";
    return 0;
}
