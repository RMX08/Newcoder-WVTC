#include <iostream>
using namespace std;

int main() {
    int a, b, c;
    cin >> a >> b >> c;

    int max = a, min = a;
    max = b>max ? b : max;
    max = c>max ? c : max;
    min = b<min ? b : min;
    min = c<min ? c : min;

    cout << "The maximum number is : " << max << endl << "The minimum number is : " << min ;

    return 0;
}
